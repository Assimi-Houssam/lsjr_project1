hostname -I
# or
ip -4 addr show

# run as Administrator
netsh interface portproxy add v4tov4 listenaddress=0.0.0.0 listenport=3000 connectaddress=<WSL_IP> connectport=3000

# allow through Windows Firewall
New-NetFirewallRule -DisplayName "Vite 3000" -Direction Inbound -Protocol TCP -LocalPort 3000 -Action Allow

#verify
netsh interface portproxy show all
netstat -an | findstr 3000

#to remove proxy
netsh interface portproxy delete v4tov4 listenport=3000 listenaddress=0.0.0.0
Remove-NetFirewallRule -DisplayName "Vite 3000"
